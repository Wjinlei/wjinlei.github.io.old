    Golang Raylib 窗口的创建与背景设置
    2022-07-03 14:56:46    golang  raylib  游戏引擎 

    简介

    Raylib是一个简单易用的跨平台的游戏绘图库，Raylib 有+60多种语言的绑定

    Hello World

    Raylib的Go语言版本的Hello World非常简单

    package main

import "github.com/gen2brain/raylib-go/raylib"

func main() {
	rl.InitWindow(800, 450, "raylib [core] example - basic window")
	rl.SetTargetFPS(60)

	// 游戏的主循环
	for !rl.WindowShouldClose() {
		rl.BeginDrawing()
		rl.ClearBackground(rl.RayWhite)
		rl.DrawText("Congrats! You created your first window!", 190, 200, 20, rl.LightGray)
		rl.EndDrawing()
	}
	rl.CloseWindow()
}
    • rl.InitWindow() 用于初始化一个窗口
    • rl.SetTargetFPS(60) 用于设置窗口的刷新率
    • rl.WindowShouldClose() 返回窗口是否被关闭的状态
    • for 循环中是窗口的绘图逻辑，也是游戏的主循环
    • rl.BeginDrawing() 开始绘图
    • rl.ClearBackground(rl.RayWhite) 用于设置背景为白色
    • rl.DrawText() 用于向窗口中绘制一段文本
    • rl.EndDrawing() 结束绘图
    • rl.CloseWindow() 销毁窗口的内存占用

    可以参考API 或 cheatsheet

    设置窗口背景

    让我们将代码重构一下，

    将屏幕的宽度和高度抽取成常量 screenWidth 和 screenHeight，

    再设置一个 running 变量用于表示游戏是否还在运行 bgkColor 用于设置背景颜色。

    将主循环中要做的事情分成三类 handleInput() 用于处理用户输入 update() 用于更新状态 render() 用于渲染场景，

    我们在 render() 函数中将背景颜色设置为 bkgColor 并且调用 drawScene() 函数，这个函数用于渲染 场景。

    在 init() 函数中设置窗口的基本属性，在 main() 函数中执行主循环。

    package main

// Create Window and set Background Color

import "github.com/gen2brain/raylib-go/raylib"

const (
	screenWidth  = 1000
	screenHeight = 400
)

var (
	running  = true
	bkgColor = rl.NewColor(147, 211, 196, 255)
)

func handleInput() {}

func update() {
	running = !rl.WindowShouldClose()
}

func drawScene() {}

func render() {
	rl.BeginDrawing()
	rl.ClearBackground(bkgColor)
	drawScene()
	rl.EndDrawing()
}

func init() {
	rl.InitWindow(screenWidth, screenHeight, "learn-raylib-go")
	rl.SetTargetFPS(60)
	rl.SetExitKey(0) // 设置退出按钮的key，0表示取消(或者说不响应退出按钮)
}

func quit() {
	rl.CloseWindow()
}

func main() {
	for running {
		handleInput()
		update()
		render()
	}
	quit()
}
