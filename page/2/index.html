    Go 游戏开发学习笔记(四)-创建相机 本文共 251 字，阅读约 2 分钟
    2022-07-04 21:36:09 golang raylib 游戏引擎

    在之前的小节中，我们创建了一个精灵，并且可以控制精灵的移动，但我们的屏幕并没有随着精灵的移动而移动，这个所谓的屏幕就是相机

    所以，这一节，我们将创建一个跟随人物移动而移动的相机

    首先，我们创建一个变量 cam 它代表一个相机对象，它的类型是 rl.Camera2D

    然后我们在 init() 函数中创建这个相机，然后在 update() 函数中更新相机的目标位置，最后在 render() 中2D模式

    package main

import "github.com/gen2brain/raylib-go/raylib"

const (
	screenWidth  = 1000
	screenHeight = 400
)

var (
	running      = true
	bkgColor     = rl.NewColor(147, 211, 196, 255)
	grassSprite  rl.Texture2D
	playerSprite rl.Texture2D
	playerSrc    rl.Rectangle
	playerDest   rl.Rectangle
	playerSpeed  float32 = 3
	musicPaused  bool
	music        rl.Music
	cam          rl.Camera2D
)

func handleInput() {
	if rl.IsKeyDown(rl.KeyW) || rl.IsKeyDown(rl.KeyUp) {
		playerDest.Y -= playerSpeed
	}
	if rl.IsKeyDown(rl.KeyD) || rl.IsKeyDown(rl.KeyRight) {
		playerDest.X += playerSpeed
	}
	if rl.IsKeyDown(rl.KeyS) || rl.IsKeyDown(rl.KeyDown) {
		playerDest.Y += playerSpeed
	}
	if rl.IsKeyDown(rl.KeyA) || rl.IsKeyDown(rl.KeyLeft) {
		playerDest.X -= playerSpeed
	}
	if rl.IsKeyPressed(rl.KeyQ) {
		musicPaused = !musicPaused
	}
}

func update() {
	running = !rl.WindowShouldClose()
	rl.UpdateMusicStream(music)
	if !musicPaused {
		rl.ResumeMusicStream(music)
	} else {
		rl.PauseMusicStream(music)
	}

	// 更新相机目标
	cam.Target = rl.NewVector2(float32(playerDest.X-(playerDest.Width/2)), float32(playerDest.Y-(playerDest.Height/2)))
}

func drawScene() {
	rl.DrawTexture(grassSprite, 100, 50, rl.White)
	rl.DrawTexturePro(playerSprite, playerSrc, playerDest,
		rl.NewVector2(playerDest.Width, playerDest.Height), 0, rl.White)
}

func render() {
	rl.BeginDrawing()
	rl.ClearBackground(bkgColor)
	rl.BeginMode2D(cam) // 开始相机
	drawScene()
	rl.EndMode2D()      // 结束相机
	rl.EndDrawing()
}

func init() {
	rl.InitWindow(screenWidth, screenHeight, "learn-raylib-go")
	rl.SetTargetFPS(60)
	rl.SetExitKey(0)
	grassSprite = rl.LoadTexture("./res/Tilesets/Grass.png")
	playerSprite = rl.LoadTexture("./res/Characters/Basic Charakter Spritesheet.png")
	playerSrc = rl.NewRectangle(0, 0, 48, 48)
	playerDest = rl.NewRectangle(200, 200, 100, 100)

	rl.InitAudioDevice()
	music = rl.LoadMusicStream("./res/Avery's Farm Loopable.mp3")
	musicPaused = false
	rl.PlayMusicStream(music)

	// 在一个2D平面中，一个矩形由两个点确定，创建相机时，需要明确这两个点，第一个点是相机的偏移量，第二个点是相机的目标，当旋转或缩放时，以目标点为中心，旋转或缩放
	// 创建一个相机，第一个参数是相机的偏移量，第二个参数是相机的目标点，第三个参数是相机的旋转角度，最后一个参数是相机的缩放
	// 这里相机的目标以人物的中心点为准
	cam = rl.NewCamera2D(
		rl.NewVector2(float32(screenWidth/2), float32(screenHeight/2)),
		rl.NewVector2(float32(playerDest.X-(playerDest.Width/2)), float32(playerDest.Y-(playerDest.Height/2))),
		0.0, 1.0)
}

func quit() {
	rl.UnloadTexture(grassSprite)
	rl.UnloadTexture(playerSprite)
	rl.UnloadMusicStream(music)
	rl.CloseAudioDevice()
	rl.CloseWindow()
}

func main() {
	for running {
		handleInput()
		update()
		render()
	}
	quit()
}
    • rl.NewCamera2D(offset, target Vector2, rotation, zoom float32) Camera2D 创建一个2D相机
    • rl.BeginMode2D(camera Camera2D) 使用自定义相机初始化2D模式
    • rl.EndMode2D() 结束2D模式
    • rl.NewVector2(x, y float32) Vector2 返回一个新的2D坐标点

    Java 字符串 本文共 966 字，阅读约 5 分钟
    2021-11-18 00:41:46 java

    字符串是不可变的

    • 为什么字符串是不可变的?
      • 为了线程安全(不可变代表着它是线程安全的)
      • 为了存储安全

      例如 HashMap,我们知道HashMap在存储的时候,会给元素计算一个hashCode值,假如你存储了一个字符串对象,
      如果你修改了它,那么这个hashCode就和原来的值不一致了,这会违反了Object类中关于hashCode的约定,
      因此为了我们能够在哈希桶中安全的使用String,字符串必须不可变

    • 不可变也导致了每当修改的时候都需要重复创建新的对象
      • 因此请不要在循环中创建字符串对象

    算法

    关于算法复杂度

    • O(1) 常数时间
    无论问题的规模有多大,总能在固定次操作后就能解决问题,代表算法有哈希桶,数组的随机寻址
    • O(n) 线性时间
    解决问题的时间和问题规模成正比,问题规模越大,需要的时间越长,代表算法是遍历
    • O(log(n)) 对数时间
    解决问题的时间和问题的对数成正比,每1次操作,解决问题的时间成倍减少,代表算法有二分查找,二叉树

    Java 8 Stream

    • 一个流
    • 简化代码
    • 不易出错
    • 可读性/可维护性++

    什么是流

    你可以把它想象成一条水流,它会把集合中的元素一个一个放入这条水流中
    无论集合是有序的还是无序的,它们在流中都会一个一个的按照顺序被取出
    你可以对取出的每一个元素做无限次数的中间处理和一次终结操作
    终结操作后,这个流就被销毁了

    Thread

    • 线程可以提高程序的执行效率
    • Java中只有这么一种东西代表线程(Thread)
    • start方法才能并发执行
    • 方法栈是线程私有的(局部变量)

    • 对于IO密集型场景及其有用
